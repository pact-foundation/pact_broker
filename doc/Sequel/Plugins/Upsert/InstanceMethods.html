def upsert(opts = {})
  @upsert_plugin_upserting = true
  if postgres? || mysql?
    save(opts)
  else
    manual_upsert(opts)
  end
  load_values_from_previously_inserted_object unless id
  self
rescue Sequel::NoExistingObject
  load_values_from_previously_inserted_object
  self
ensure
  @upsert_plugin_upserting = false
end
